var lista = ["maça", "pera", "laranja"]

lista.push() -> adiciona a lista
lista.pop()  -> retira o último elemento da lista
lista.length -> mostra o tamanho da 
lista.reverse() -> mostra a lista de trás para frente
lista.toString() -> transforma a lista em uma string
lista.join(-) -> mosta a string separando as palavras pos -
